56std::pair<double,int> get_chi2 ( std::vector<unsigned int>& vec )
57{
58 if (vec.empty())
59 return {-1,-1};
60
61 auto min_val = *std::min_element(vec.begin(), vec.end()); // Наименьший элемент выборки
62 auto max_val = *std::max_element(vec.begin(), vec.end()); // Наибольший элемент выборки
63
64 int bin_number = 1 + std::log2(vec.size()); // Количество промежутков по правилу Стерджеса
65 float bin_width = (float)(max_val-min_val)/bin_number; // Ширина промежутка
66
67 float expected = (float)vec.size() / bin_number; // Количество ожидаемых наблюдений в каждом промежутке
68 std::vector<int> observed(bin_number, 0); // Количество фактических наблюдений в каждом промежутке
69
70
71 for ( auto val : vec )
72 {
73 int bin_ind = (int) ((val - min_val) / bin_width);
74 if (bin_ind >= bin_number) --bin_ind;
75
76 ++observed.at(bin_ind);
77 }
78
79
80 double chi2 = 0;
81
82 for ( auto obs : observed)
83 {
84 chi2 += (obs - expected) * (obs - expected) / expected;
85 }
86
87 return {chi2, bin_number-1};
