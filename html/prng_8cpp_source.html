61uint32_t mid_xor_PRNG::generate ()
62{
63 uint64_t product = seed * seed;
64
65 seed = (product >> 8) & 0xFFFFFFFF;
66
67 // Динамический сдвиг для XorShift
68 uint8_t shift = (seed >> 10) & 0x1F; // Сдвиг от 0 до 31
69 seed ^= 0x9E3779B9 << (shift % 13 + 1); // Чтобы не было вырождения в 0
70 seed ^= seed >> (shift % 17 + 1);
71
72 return result(seed);
73}
