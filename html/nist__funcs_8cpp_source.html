18std::vector<bool> sample_to_bit_sequence ( std::vector<unsigned int>& vec )
19{
20 int max_len = 1 + std::log2(MAX_LIM); // Находим наибольший бит, который может принять значение 1
21 // Старшие биты всегда будут 0, что будет нарушать тесты, поэтому обрубаем их
22
23 std::vector<bool> bit_seq;
24
25 for (auto num : vec)
26 {
27 for (int i=0; i < max_len; ++i)
28 {
29 bit_seq.push_back(num >> i & 1); // биты хранятся в обратном порядке, т.е. чем меньше индекс, тем младше бит
30 }
31 }
32
33 return bit_seq;

43std::vector<unsigned int> get_pattern_counts ( std::vector<bool>& bit_seq, int m )
44{
45 std::vector<unsigned int> pattern_counts(1<<m, 0);
46
47 for (int i=0; i < bit_seq.size(); ++i)
48 {
49 int k=0;
50 for (int j = m-1; j >= 0; --j) // Проходимся в обратном порядке для эстетичности, чтобы паттерн равный числу I был записан под индексом I
51 {
52 if (bit_seq.at((i+j) % bit_seq.size())) k = 2*k + 1;
53 else k = 2*k;
54 }
55 ++pattern_counts[k];
56 }
57
58 return pattern_counts;
59}

66double get_psi2 ( std::vector<bool>& bit_seq, int m )
67{
68 std::vector<unsigned int> pattern_counts = get_pattern_counts(bit_seq, m);
69
70
71 int expected = bit_seq.size() / (1 << m);
72
73 double sum = 0;
74 for ( auto count : pattern_counts)
75 {
76 sum += (count - expected) * (count - expected);
77 }
78
79 sum = ( sum * (1 << m) ) / bit_seq.size() - bit_seq.size();
80
81 return sum;
82}
