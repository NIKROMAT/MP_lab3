18cephes_igamc(double a, double x)
19{
20 double ans, ax, c, yc, r, t, y, z;
21 double pk, pkm1, pkm2, qk, qkm1, qkm2;
22
23 if ( (x <= 0) || ( a <= 0) )
24 return( 1.0 );
25
26 if ( (x < 1.0) || (x < a) )
27 return( 1.e0 - cephes_igam(a,x) );
28
29 ax = a * log(x) - x - cephes_lgam(a);
30
31 if ( ax < -MAXLOG ) {
32 printf("igamc: UNDERFLOW\n");
33 return 0.0;
34 }
35 ax = exp(ax);
36
37 /* continued fraction */
38 y = 1.0 - a;
39 z = x + y + 1.0;
40 c = 0.0;
41 pkm2 = 1.0;
42 qkm2 = x;
43 pkm1 = x + 1.0;
44 qkm1 = z * x;
45 ans = pkm1/qkm1;
46
47 do {
48 c += 1.0;
49 y += 1.0;
50 z += 2.0;
51 yc = y * c;
52 pk = pkm1 * z - pkm2 * yc;
53 qk = qkm1 * z - qkm2 * yc;
54 if ( qk != 0 ) {
55 r = pk/qk;
56 t = fabs( (ans - r)/r );
57 ans = r;
58 }
59 else
60 t = 1.0;
61 pkm2 = pkm1;
62 pkm1 = pk;
63 qkm2 = qkm1;
64 qkm1 = qk;
65 if ( fabs(pk) > big ) {
66 pkm2 *= biginv;
67 pkm1 *= biginv;
68 qkm2 *= biginv;
69 qkm1 *= biginv;
70 }
71 } while ( t > MACHEP );
72
73 return ans*ax;
74}

76double
77cephes_igam(double a, double x)
78{
79 double ans, ax, c, r;
80
81 if ( (x <= 0) || ( a <= 0) )
82 return 0.0;
83
84 if ( (x > 1.0) && (x > a ) )
85 return 1.e0 - cephes_igamc(a,x);
86
87 /* Compute x**a * exp(-x) / gamma(a) */
88 ax = a * log(x) - x - cephes_lgam(a);
89 if ( ax < -MAXLOG ) {
90 printf("igam: UNDERFLOW\n");
91 return 0.0;
92 }
93 ax = exp(ax);
94
95 /* power series */
96 r = a;
97 c = 1.0;
98 ans = 1.0;
99
100 do {
101 r += 1.0;
102 c *= x/r;
103 ans += c;
104 } while ( c/ans > MACHEP );
105
106 return ans * ax/a;
107}

146 int i;

152 w = cephes_lgam(q); /* note this modifies sgngam! */

155lgsing:
156 goto loverf;

160 sgngam = -1;
161 else

164 if ( z > 0.5 ) {
165 p += 1.0;
166 z = p - q;
167 }
168 z = q * sin( PI * z );
169 if ( z == 0.0 )
170 goto lgsing;

172 z = log(PI) - log( z ) - w;

178 p = 0.0;

181 p -= 1.0;
182 u = x + p;
183 z *= u;
184 }
185 while ( u < 2.0 ) {
186 if ( u == 0.0 )
187 goto lgsing;

193 sgngam = -1;
194 z = -z;
195 }

266cephes_erf(double x)
267{
268 static const double two_sqrtpi = 1.128379167095512574;
269 double sum = x, term = x, xsqr = x * x;
270 int j = 1;
271
272 if ( fabs(x) > 2.2 )
273 return 1.0 - cephes_erfc(x);
274
275 do {
276 term *= xsqr/j;
277 sum -= term/(2*j+1);
278 j++;
279 term *= xsqr/j;
280 sum += term/(2*j+1);
281 j++;
282 } while ( fabs(term)/sum > rel_error );

288cephes_erfc(double x)
289{
290 static const double one_sqrtpi = 0.564189583547756287;
291 double a = 1, b = x, c = x, d = x*x + 0.5;
292 double q1, q2 = b/d, n = 1.0, t;
293
294 if ( fabs(x) < 2.2 )
295 return 1.0 - cephes_erf(x);
296 if ( x < 0 )
297 return 2.0 - cephes_erfc(-x);
298
299 do {
300 t = a*n + b*x;
301 a = b;
302 b = t;
303 t = c*n + d*x;
304 c = d;
305 d = t;
306 n += 0.5;
307 q1 = q2;
308 q2 = b/d;
309 } while ( fabs(q1-q2)/q2 > rel_error );
310
311 return one_sqrtpi*exp(-x*x)*q2;
