29double nist_frequency ( std::vector<bool>& bit_seq )
30{
31 int sum = 0;
32
33 for (auto bit : bit_seq)
34 {
35 bit ? ++sum : --sum; // Итого: ++sum, если бит - 1, иначе --sum
36 }
37
38 double s_obs = std::abs(sum) / std::sqrt(bit_seq.size());
39 double p_value = std::erfc(s_obs / std::sqrt(2));
40
41 return p_value;
42}

48double nist_runs ( std::vector<bool>& bit_seq )
49{
50
51 int sum = std::count(bit_seq.begin(), bit_seq.end(), true);
52
53 double pi = (double)sum / bit_seq.size();
54
55 if (fabs(pi - 0.5) >= 2 / std::sqrt(bit_seq.size()))
56 {
57 std::cerr << ">>> Не выполнен критерий для runs теста" << std::endl;
58 return 0;
59 }
60
61
62 int runs = 1;
63 for (int i=1; i < bit_seq.size(); ++i)
64 {
65 if (bit_seq.at(i) != bit_seq.at(i-1))
66 ++runs;
67 }
68
69 double erfc_arg = fabs(runs - 2.0 * bit_seq.size() * pi * (1-pi)) / (2.0 * pi * (1-pi) * sqrt(2*bit_seq.size()));
70 double p_value = erfc(erfc_arg);
71
72 return p_value;
73
74}

94double nist_apEntropy ( std::vector<bool>& bit_seq, int m )
95{
96 double apEn_m = get_apEn(bit_seq, m);
97 double apEn_m1 = get_apEn(bit_seq, m+1);
98
99 double apEn = apEn_m - apEn_m1;
100 double chi2 = 2 * bit_seq.size() * (std::log(2) - apEn);
101
102 double p_value = cephes_igamc(1 << (m-1), chi2/2);
103 return p_value;
104}

109double nist_cusum ( std::vector<bool>& bit_seq )
110{

113
114

117 bit ? ++S : --S;
118 max_S = (std::abs(S) > max_S) ? std::abs(S) : max_S;
