72template <class PRNG>
73void create_samples (std::string path, std::string prng_name,
74 std::vector<int>& sizes, std::vector<int>& seeds,
75 uint32_t min_lim, uint32_t max_lim)
76{
77
78
79 int mod;
80
81 // Проверка корректности размеров списков seeds и sizes
82 if (sizes.size() == seeds.size()) mod = sizes.size();
83 else if (sizes.size() % seeds.size() == 0) mod = sizes.size() / seeds.size();
84 else
85 {
86 std::cerr << "! Неправильные размерности набора размеров и семян генераторов." << std::endl;
87 return;
88 }
89
90
91 std::cout << "Generating samples by " + prng_name << std::endl;
92
93 // Создание папки для выборок
94 fs::create_directory(path + prng_name);
95
96 // Цикл создания выборок
97 for (int i=0; i < sizes.size(); ++i)
98 {
99
100 // Создать файл выборки
101 std::ofstream sample_file(path + prng_name + "/" + prng_name + "_" + std::to_string(i+1) + ".txt");
102
103 // Создать генератор с заданным семенем и границами
104 PRNG prng (seeds.at(i % mod), min_lim, max_lim);
105
106 // Создать и записать выборку
107 prng.generate_sample(sample_file, sizes.at(i));
108
109 // Закрыть файл выборки
110 sample_file.close();
111
112 std::cout << "> Complete " + prng_name + "_" + std::to_string(i+1) + ".txt" << std::endl;
113 }
114
115 std::endl(std::cout);
116}
